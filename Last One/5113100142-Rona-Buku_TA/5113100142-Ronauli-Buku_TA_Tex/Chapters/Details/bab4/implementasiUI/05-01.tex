/** 
 * File : app/Http/Controllers/UserController
 * Menampilkan halaman daftar pengguna
 * langsung diFetch dari base Model User
 * Method : GET
 */
public function user()
{
    $data['data'] = User::paginate(20);
    return view('pages.user.index', $data);
}
